	$(document).ready(function(){

        $('.search-header #searchAutosuggestWrapper').on('touchstart, touchend', function() {
            $('body').addClass('search-engaged');
            $('.search-entry').addClass('mobile-search');
        });

		/* Any click/tap to dis-engage sidebar */
		$(document).click(function() {
			$('.search-header-left').show();

			if (autosuggestSearchInput.is(":focus")) {
				$('body').addClass('search-engaged');

				if ($('body').innerWidth() < 768) {
					$('.search-entry').addClass('mobile-search');
				}
			} else {
				$('body').removeClass('search-engaged');
			}
		});

		/* Add esc press to leave search-engaged */
        $(document).keyup(function(e) {
            if (autosuggestSearchInput.is(":focus")) {
                if (e.keyCode === 27) { //27 is esc key
                    $('body').removeClass('search-engaged');
                    autosuggestSearchInput.blur();
                }
            }
        });

		$('.collapsible-search').click(function () {
			var id = this.id;
			$('.' + id + '_list').toggleClass('open');
			$('#arrow_' + id).toggleClass('open');
		});

		$('.prof-class-menu .dropdown-item').click(function() {

			var profClassData = $(this).data();

			var previous = $('.prof-class-menu .dropdown-item.selected');
			previous.removeClass('selected');
			$(this).addClass('selected');

			// Set button text
			var profClassSelector = $('#profClassSelectText');
			profClassSelector.text(profClassData['profClassDisplay']);

			$('#searchField').val('');

			var autosuggest = $('#autosuggestContainer').data('autosuggest');
			autosuggest.options.requestUrl = profClassData['suggestionsEndpoint'];
			autosuggest.options.resultsUrl = profClassData['resultsEndpoint'];
			autosuggest.clearSuggestions();
		});
	});
