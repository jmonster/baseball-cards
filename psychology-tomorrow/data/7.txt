		$(document).click(function() {
			$('.search-header-left').show();

			if (autosuggestSearchInput.is(":focus")) {
				$('body').addClass('search-engaged');

				if ($('body').innerWidth() < 768) {
					$('.search-entry').addClass('mobile-search');
				}
			} else {
				$('body').removeClass('search-engaged');
			}
		});

		/* Add esc press to leave search-engaged */
        $(document).keyup(function(e) {
            if (autosuggestSearchInput.is(":focus")) {
                if (e.keyCode === 27) { //27 is esc key
                    $('body').removeClass('search-engaged');
                    autosuggestSearchInput.blur();
                }
            }
        });

		$('.collapsible-search').click(function () {
			var id = this.id;
			$('.' + id + '_list').toggleClass('open');

		$('.prof-class-menu .dropdown-item').click(function() {

			var profClassData = $(this).data();

		});
	});
